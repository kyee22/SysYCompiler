int a=10,b=10;

	if(a == 1) {
		continue;		// 7
	}
	for(; a > 0; ) {
		a = a + 1;
		if(a == 10) break;
		else continue;
	}
	return 0;
}

			}
			if(a % 2 == 1) continue;
		}
